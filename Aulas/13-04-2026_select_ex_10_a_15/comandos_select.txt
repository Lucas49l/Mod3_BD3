10: vendedores faixa comissão = a ou b
SELECT * FROM vendedor 
WHERE faixa_comiss IN ("a", "b");

11: endereço = null
SELECT * FROM cliente 
WHERE endereco IS NULL;

12: vendedores e salarios fixos, ordem alfabetica vendedores
SELECT nome_vend, sal_fixo 
FROM vendedor 
ORDER BY nome_vend, 1 ASC;

SELECT nome_vend, sal_fixo, (120.00 + (sal_fixo * 0.75)) as novo_salario_fixo , faixa_comiss

WHERE faixa_comiss = 'c'
